class Solution {
    public int[] getConcatenation(int[] nums) {
        int n = nums.length;
        int[] ans = new int[2 * n];  // allocate space for concatenated array
        
        for (int i = 0; i < n; i++) {
            ans[i] = nums[i];       // assign to first half
            ans[i + n] = nums[i];   // assign to second half
        }
        
        return ans;
    }
}
